#include <stdio.h>

// Function to calculate the Nth Fibonacci number using bottom-up DP
int fibonacciBottomUp(int n) {
    if (n <= 1) {
        return n; 
    }

    int dp[n + 1];
    dp[0] = 0;
    dp[1] = 1;
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}

int main() {
    int n;
    printf("Enter a non-negative integer: ");
    scanf("%d", &n);

    if (n < 0) {
        printf("Please enter a non-negative integer.\n");
    } else {
        printf("The %dth Fibonacci number (bottom-up) is: %d\n", n, fibonacciBottomUp(n));
    }

    return 0;
}
